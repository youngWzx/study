function hashMeshIdToBroadcast(MeshId) {
  var id = parseInt(MeshId);
  var temp = id;
  var magic = 'skytone';
  for (var i = 0; i < 16; i++) {
    id ^=magic.charCodeAt([i % 7]);
    temp = id >> 24;
    id <<= 8;
    id ^=temp;
  }
  return id;
}
